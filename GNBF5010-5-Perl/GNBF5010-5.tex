\begin{frame}[fragile]{Perl One Liners}
\begin{block}{Options}
  \begin{itemize}
    \item -e
    \item -n
    \begin{verbatim}
while (<>) {
  # your code goes here
}
    \end{verbatim}
    \item -p
\begin{verbatim}
  while (<>) {
      # your code goes here
    } continue {
      print or die "-p destination: $!\n";
    }
\end{verbatim}
  \end{itemize}
